# Layout Priorities

Content Compression Resistance Priority

Content Hugging Priority

使用在一个水平面上的两个 `UILabel` 来作为演示代码说明。

```
override init(style: UITableViewCellStyle, reuseIdentifier: String?) {
    super.init(style: style, reuseIdentifier: reuseIdentifier)
    
    self.contentView.addSubview(self.checkButton)
    self.contentView.addSubview(self.avatarImageView)
    self.contentView.addSubview(self.nameLabel)
    self.contentView.addSubview(self.subtitleLabel)
    
    let marginGuide = self.contentView.layoutMarginsGuide
    
    let checkButtonSize: CGFloat = 24
    self.checkButton.widthAnchor.constraint(equalToConstant: checkButtonSize)
        .isActive = true
    self.checkButton.heightAnchor.constraint(equalToConstant: checkButtonSize)
        .isActive = true
    self.checkButton.leadingAnchor.constraint(equalTo: marginGuide.leadingAnchor)
        .isActive = true
    self.checkButton.centerYAnchor.constraint(equalTo: marginGuide.centerYAnchor)
        .isActive = true
    
    let avatarImageSize: CGFloat = 24
    self.avatarImageView.widthAnchor.constraint(equalToConstant: avatarImageSize)
        .isActive = true
    self.avatarImageView.heightAnchor.constraint(equalToConstant: avatarImageSize)
        .isActive = true
    self.avatarImageView.leadingAnchor.constraint(equalTo: self.checkButton.trailingAnchor, constant: 10)
        .isActive = true
    self.avatarImageView.centerYAnchor.constraint(equalTo: marginGuide.centerYAnchor)
        .isActive = true
    
    self.nameLabel.leadingAnchor.constraint(equalTo: self.avatarImageView.trailingAnchor, constant: 10)
        .isActive = true
    self.nameLabel.trailingAnchor.constraint(equalTo: self.subtitleLabel.leadingAnchor, constant: -10)
        .isActive = true
    self.nameLabel.heightAnchor.constraint(equalToConstant: 17)
        .isActive = true
    self.nameLabel.centerYAnchor.constraint(equalTo: marginGuide.centerYAnchor)
        .isActive = true
    self.nameLabel.setContentHuggingPriority(UILayoutPriority.defaultLow + 1,
                                             for: .horizontal)
    self.nameLabel.setContentCompressionResistancePriority(UILayoutPriority.defaultHigh + 1,
                                                           for: .horizontal)
    
    self.subtitleLabel.leadingAnchor.constraint(equalTo: self.nameLabel.trailingAnchor, constant: 10)
        .isActive = true
    self.subtitleLabel.trailingAnchor.constraint(equalTo: marginGuide.trailingAnchor)
        .isActive = true
    self.subtitleLabel.heightAnchor.constraint(equalToConstant: 17)
        .isActive = true
    self.subtitleLabel.centerYAnchor.constraint(equalTo: marginGuide.centerYAnchor)
        .isActive = true
}
```
